import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

np.random.seed(42)

departments = ['R&D','Marketing','Sales','Finance','HR','IT','Operations','Legal']
regions = ['North America','Asia Pacific','Europe','Latin America','Africa','Middle East']

n = 100
employee_ids = [f"EMP{str(i+1).zfill(3)}" for i in range(n)]
dept_choices = np.random.choice(departments, size=n)
region_choices = np.random.choice(regions, size=n)
performance_scores = np.round(np.random.normal(80, 8, n), 2)
years_experience = np.random.randint(1, 31, n)
satisfaction_rating = np.round(np.random.uniform(3.5, 5.0, n), 1)

df = pd.DataFrame({
    'employee_id': employee_ids,
    'department': dept_choices,
    'region': region_choices,
    'performance_score': performance_scores,
    'years_experience': years_experience,
    'satisfaction_rating': satisfaction_rating
})

# Frequency for IT
it_count = (df['department'] == 'IT').sum()
print("Frequency count for IT department:", it_count)

# Histogram
df['department'].value_counts().plot(kind='bar')
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.show()
